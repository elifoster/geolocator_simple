def get(ip)
  return @cache[ip] if @cache.key?(ip)
  url = format(URL_FORMAT, ip)
  response = Curl.get(url).body_str
  ipapi = SimpleGeolocator::IPAPIResponse.new(Oj.load(response))
  @cache[ip] = ipapi
end
